const canvas = document.getElementById('usCanvas'); const ctx = canvas.getContext('2d'); const mapImage = new Image(); mapImage.src = 'blank_US_states.png'; // US geographic bounds including Alaska and Hawaii const NORTH = 71.5; const SOUTH = 18.5; const WEST = -170.0; const EAST = -65.0; let CANVAS_W = 0; let CANVAS_H = 0; let mapLoaded = false; let latToY, lonToX, pixelToLatLon; mapImage.onload = () => { CANVAS_W = mapImage.width; CANVAS_H = mapImage.height; canvas.width = CANVAS_W; canvas.height = CANVAS_H; // Define functions using actual canvas dimensions latToY = function(lat) { return ((NORTH - lat) / (NORTH - SOUTH)) * CANVAS_H; }; lonToX = function(lon) { return ((lon - WEST) / (EAST - WEST)) * CANVAS_W; }; pixelToLatLon = function(x, y) { return { lat: NORTH - (y / CANVAS_H) * (NORTH - SOUTH), lon: WEST + (x / CANVAS_W) * (EAST - WEST) }; }; mapLoaded = true; drawMap(); initGame(); }; mapImage.onerror = () => { document.getElementById('message').innerHTML = "⚠️ Upload blank_US_states.png"; };
